    from browser import timer

    print("¡Bienvenido!")
    numero = 0

    # Definimos lo que hará el boton cuando sea apretado
    def adivinar():
    
        global numero
        numero = int(input("Adivine el número: "))
        if numero == 42:
            print("¡Ganaste!")
        else:
            if numero > 42:
                print("Alto")
            else:
                print("Bajo")
        
        if numero != 42:
            # Si el número no se adivinó, se repite la función después de
            # 3 segundos 
            timer.set_timeout(adivinar, 3000)

    adivinar()
