            ``ends_equal(["aba", "xyz", "aa", "x", "bbb"])`` -> ``3`` |br|

            class myTests(TestCaseGui):
                def testOne(self):

            ~~~~
            def x_before(words):

                
            ====
            from unittest.gui import TestCaseGui


            class myTests(TestCaseGui):
                def testOne(self):
            
                    self.assertEqual(
                        x_before(["bbb", "ccc", "axx", "xzz", "xaa"]),

                    )

                        x_before(["ccc", "bbb", "aaa", "xcc", "xaa"]),
